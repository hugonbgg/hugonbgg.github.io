// Define two arrays
let a = [100, 200];
let b = [[0, 3], [0, 4], [1, 5], [2, 5]];

// Define empty array for result
let result = [];

// Loop over 'b', each time adding next element in 'result'
for(let i in b) {
    result.push([a, b[i]]);
}

// In the end, 'result' contains the following array:
// [[[100,200],[0,3]],[[100,200],[0,4]],[[100,200],[1,5]],[[100,200],[2,5]]]
